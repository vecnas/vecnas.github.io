Быстрый старт

Создаем страницу 

  <html>
  <head>
    <script type="text/javascript" src="https://ajax.googleapis.com/ajax/libs/jquery/1.9.0/jquery.js"></script>
    <script type="text/javascript" src='jiant.js'></script>
  </head>
  <body>

    <div id="_mainView">
      <div class="_label"></div>
      <button class="_ctl">Say hello</button>
    </div>
  </body>
  </html>
и скрипт 

  // определение приложения
  var hello = {
    id: "hello",
    views: {
      mainView: {
        label: jiant.label,
        ctl: jiant.ctl
      }
    }
  };

  $(function () {

    // связываем приложение с HTML, включаем режим разработки
    jiant.bindUi("_", hello, true);

    // функционал
    jiant.onUiBound(hello, function($, app) {
        var view = app.views.mainView;
        view.ctl.click(function() {
        view.label.html("Hello!");
        });
    });

  });
